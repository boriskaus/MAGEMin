		}

	}
 
	
	for (k = 0; k < d->z_b.nzEl_val; k++){

   for (int i = 0; i < SS_ref_db.n_xeos; i++){

   }

   nlopt_destroy(SS_ref_db.opt);
  
   return SS_ref_db;
};
